10.Enter 5 values of array from user, enter a value from user and search this value in array, if value is found then print “found” or not then print “not found”?

#include <stdio.h>
void q10() {
    int array[5], i, key, found = 0;
    printf("Q10 - Enter 5 array elements: ");
    for (i = 0; i < 5; i++) scanf("%d", &array[i]);
    printf("Enter value to search: ");
    scanf("%d", &key);
    for (i = 0; i < 5; i++) if (array[i] == key) { found = 1; break; }
    if (found) printf("Found\n"); else printf("Not Found\n");
}
